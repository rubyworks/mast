# File lib/mast/cli.rb, line 101
    def run_command
      optparse
      if @command.size > 1
        raise ArgumentError, "Please issue only one command."
      end
      case @command.first
      when :help     then help
      when :create   then generate
      when :update   then update
      when :list     then list
      #when :show     then show
      when :diff     then diff
      when :new      then new
      when :old      then old
      when :verify   then verify
      when :clean    then clean
      else
        generate
      end
    end

    def report_help
      doc = false
      File.readlines(__FILE__).each do |line|
        line = line.strip
        break if doc && line.empty?
        next if line =~ /^#!/
        next if line.empty?
        puts line[1..-1].sub(/^\ \ /,'')
        doc = true
      end
    end
